Form Helper

This helper will help you to create form elements.

Code example:

    Liber::loadHelper('Form');

    // input
    form_input_('name', 'value', 'class="my_input"');
    


    // password
    form_password_('name', 'value', 'class="my_input"');
    


    // textarea
    form_textarea_('name', 'value', 'class="my_input"');
    


    // button
    form_button_('name', 'button', 'class="my_button"');
    


    // select
    form_select_('name', Array('T'=>'True', 'F'=>'False'), Array('F'), 'class="my_select"');
