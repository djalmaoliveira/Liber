 37: /**
 38:  * Compress HTML
 39:  *
 40:  * This is a heavy regex-based removal of whitespace, unnecessary comments and
 41:  * tokens. IE conditional comments are preserved. There are also options to have
 42:  * STYLE and SCRIPT blocks compressed by callback functions.
 43:  *
 44:  * A test suite is available.
 45:  *
 46:  * @package Minify
 47:  * @author Stephen Clay <steve@mrclay.org>
 48:  */
 49: class Minify_HTML {
 50: 
 51:     /**
 52:      * "Minify" an HTML page
 53:      *
 54:      * @param string $html
 55:      *
 56:      * @param array $options
 57:      *
 58:      * 'cssMinifier' : (optional) callback function to process content of STYLE
 59:      * elements.
 60:      *
 61:      * 'jsMinifier' : (optional) callback function to process content of SCRIPT
 62:      * elements. Note: the type attribute is ignored.
 63:      *
 64:      * 'xhtml' : (optional boolean) should content be treated as XHTML1.0? If
 65:      * unset, minify will sniff for an XHTML doctype.
 66:      *
 67:      * @return string
 68:      */
 69:     public static function minify($html, $options = array()) {
 70:         $min = new Minify_HTML($html, $options);
 71:         return $min->process();
 72:     }
 73: 
 74: 
 75:     /**
 76:      * Create a minifier object
 77:      *
 78:      * @param string $html
 79:      *
 80:      * @param array $options
 81:      *
 82:      * 'cssMinifier' : (optional) callback function to process content of STYLE
 83:      * elements.
 84:      *
 85:      * 'jsMinifier' : (optional) callback function to process content of SCRIPT
 86:      * elements. Note: the type attribute is ignored.
 87:      *
 88:      * 'xhtml' : (optional boolean) should content be treated as XHTML1.0? If
 89:      * unset, minify will sniff for an XHTML doctype.
 90:      *
 91:      * @return null
 92:      */
 93:     public function __construct($html, $options = array())
 94:     {
 95:         $this->_html = str_replace("\r\n", "\n", trim($html));
 96:         if (isset($options['xhtml'])) {
 97:             $this->_isXhtml = (bool)$options['xhtml'];
 98:         }
 99:         if (isset($options['cssMinifier'])) {
100:             $this->_cssMinifier = $options['cssMinifier'];
101:         }
102:         if (isset($options['jsMinifier'])) {
103:             $this->_jsMinifier = $options['jsMinifier'];
104:         }
105:     }
106: 
107: 
108:     /**
109:      * Minify the markeup given in the constructor
110:      *
111:      * @return string
112:      */
113:     public function process()
114:     {
115:         if ($this->_isXhtml === null) {
116:             $this->_isXhtml = (false !== strpos($this->_html, '<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML'));
117:         }
118: 
119:         $this->_replacementHash = 'MINIFYHTML' . md5($_SERVER['REQUEST_TIME']);
120:         $this->_placeholders = array();
121: 
122:         // replace SCRIPTs (and minify) with placeholders
123:         $this->_html = preg_replace_callback(
124:             '/(\\s*)(<script\\b[^>]*?>)([\\s\\S]*?)<\\/script>(\\s*)/i'
125:             ,array($this, '_removeScriptCB')
126:             ,$this->_html);
127: 
128:         // replace STYLEs (and minify) with placeholders
129:         $this->_html = preg_replace_callback(
130:             '/\\s*(<style\\b[^>]*?>)([\\s\\S]*?)<\\/style>\\s*/i'
131:             ,array($this, '_removeStyleCB')
132:             ,$this->_html);
133: 
134:         // remove HTML comments (not containing IE conditional comments).
135:         $this->_html = preg_replace_callback(
136:             '/<!--([\\s\\S]*?)-->/'
137:             ,array($this, '_commentCB')
138:             ,$this->_html);
139: 
140:         // replace PREs with placeholders
141:         $this->_html = preg_replace_callback('/\\s*(<pre\\b[^>]*?>[\\s\\S]*?<\\/pre>)\\s*/i'
142:             ,array($this, '_removePreCB')
143:             ,$this->_html);
144: 
145:         // replace TEXTAREAs with placeholders
146:         $this->_html = preg_replace_callback(
147:             '/\\s*(<textarea\\b[^>]*?>[\\s\\S]*?<\\/textarea>)\\s*/i'
148:             ,array($this, '_removeTextareaCB')
149:             ,$this->_html);
150: 
151:         // trim each line.
152:         // @todo take into account attribute values that span multiple lines.
153:         $this->_html = preg_replace('/^\\s+|\\s+$/m', '', $this->_html);
154: 
155:         // remove ws around block/undisplayed elements
156:         $this->_html = preg_replace('/\\s+(<\\/?(?:area|base(?:font)?|blockquote|body'
157:             .'|caption|center|cite|col(?:group)?|dd|dir|div|dl|dt|fieldset|form'
158:             .'|frame(?:set)?|h[1-6]|head|hr|html|legend|li|link|map|menu|meta'
159:             .'|ol|opt(?:group|ion)|p|param|t(?:able|body|head|d|h||r|foot|itle)'
160:             .'|ul)\\b[^>]*>)/i', '$1', $this->_html);
161: 
162:         // remove ws outside of all elements
163:         $this->_html = preg_replace_callback(
164:             '/>([^<]+)</'
165:             ,array($this, '_outsideTagCB')
166:             ,$this->_html);
167: 
168:         // use newlines before 1st attribute in open tags (to limit line lengths)
169:         $this->_html = preg_replace('/(<[a-z\\-]+)\\s+([^>]+>)/i', "$1\n$2", $this->_html);
170: 
171:         // fill placeholders
172:         $this->_html = str_replace(
173:             array_keys($this->_placeholders)
174:             ,array_values($this->_placeholders)
175:             ,$this->_html
176:         );
177:         return $this->_html;
178:     }
179: 
180:     protected function _commentCB($m)
181:     {
182:         return (0 === strpos($m[1], '[') || false !== strpos($m[1], '<!['))
183:             ? $m[0]
184:             : '';
185:     }
186: 
187:     protected function _reservePlace($content)
188:     {
189:         $placeholder = '%' . $this->_replacementHash . count($this->_placeholders) . '%';
190:         $this->_placeholders[$placeholder] = $content;
191:         return $placeholder;
192:     }
193: 
194:     protected $_isXhtml = null;
195:     protected $_replacementHash = null;
196:     protected $_placeholders = array();
197:     protected $_cssMinifier = null;
198:     protected $_jsMinifier = null;
199: 
200:     protected function _outsideTagCB($m)
201:     {
202:         return '>' . preg_replace('/^\\s+|\\s+$/', ' ', $m[1]) . '<';
203:     }
204: 
205:     protected function _removePreCB($m)
206:     {
207:         return $this->_reservePlace($m[1]);
208:     }
209: 
210:     protected function _removeTextareaCB($m)
211:     {
212:         return $this->_reservePlace($m[1]);
213:     }
214: 
215:     protected function _removeStyleCB($m)
216:     {
217:         $openStyle = $m[1];
218:         $css = $m[2];
219:         // remove HTML comments
220:         $css = preg_replace('/(?:^\\s*<!--|-->\\s*$)/', '', $css);
221: 
222:         // remove CDATA section markers
223:         $css = $this->_removeCdata($css);
224: 
225:         // minify
226:         $minifier = $this->_cssMinifier
227:             ? $this->_cssMinifier
228:             : 'trim';
229:         $css = call_user_func($minifier, $css);
230: 
231:         return $this->_reservePlace($this->_needsCdata($css)
232:             ? "{$openStyle}/*<![CDATA[*/{$css}/*]]>*/</style>"
233:             : "{$openStyle}{$css}</style>"
234:         );
235:     }
236: 
237:     protected function _removeScriptCB($m)
238:     {
239:         $openScript = $m[2];
240:         $js = $m[3];
241: 
242:         // whitespace surrounding? preserve at least one space
243:         $ws1 = ($m[1] === '') ? '' : ' ';
244:         $ws2 = ($m[4] === '') ? '' : ' ';
245: 
246:         // remove HTML comments (and ending "//" if present)
247:         $js = preg_replace('/(?:^\\s*<!--\\s*|\\s*(?:\\/\\/)?\\s*-->\\s*$)/', '', $js);
248: 
249:         // remove CDATA section markers
250:         $js = $this->_removeCdata($js);
251: 
252:         // minify
253:         $minifier = $this->_jsMinifier
254:             ? $this->_jsMinifier
255:             : 'trim';
256:         $js = call_user_func($minifier, $js);
257: 
258:         return $this->_reservePlace($this->_needsCdata($js)
259:             ? "{$ws1}{$openScript}/*<![CDATA[*/{$js}/*]]>*/</script>{$ws2}"
260:             : "{$ws1}{$openScript}{$js}</script>{$ws2}"
261:         );
262:     }
263: 
264:     protected function _removeCdata($str)
265:     {
266:         return (false !== strpos($str, '<![CDATA['))
267:             ? str_replace(array('<![CDATA[', ']]>'), '', $str)
268:             : $str;
269:     }
270: 
271:     protected function _needsCdata($str)
272:     {
273:         return ($this->_isXhtml && preg_match('/(?:[<&]|\\-\\-|\\]\\]>)/', $str));
274:     }
