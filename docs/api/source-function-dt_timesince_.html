23:     function dt_diffsince_($past, $future, $detail_level=0) {
24:         static $i18n = null;
25:         $lang = strtolower(Liber::conf('LANG'));
26:         if ( !is_array($i18n[$lang]) ) {
27:             $i18n[$lang] = include Liber::conf('BASE_PATH').'i18n/helper/DT.'.$lang.'.php';
28:         }
29:         $periods = &$i18n[$lang]['PERIODS'];
30:         $future = !is_numeric($future)?strtotime($future):$future;
31:         $past   = !is_numeric($past)?strtotime($past):$past;
32: 
33:         $seconds = ($future)-($past);
34:         $range = Array();
35:         $aDate = Array();
36:         $range['year']  = 31536000; //  60*60*24*365
37:         $range['month'] =  2592000; //  day*30
38:         $range['week']  =   604800; //  day * 7
39:         $range['day']   =    86400; //  hour * 24
40:         $range['hour']  =     3600; //  60*60
41:         $range['minute']=       60;
42:         $range['second']=        1;
43: 
44:         foreach($range as $id => $count) {
45:             if ( $detail_level >= 0 and $seconds >= $count ) {
46:                 $v   = floor($seconds/$count);
47:                 $aDate[$id] =  $v.' '.($v>1?$periods[$id][1]:$periods[$id][0]);
48:                 $seconds = $seconds % $count;
49:                 $detail_level--;
50:             }
51:         }
52: 
53:         if ( count($aDate) > 1 ) {
54:             $last = end($aDate);
55:             $aDate[key($aDate)] = $i18n[$lang]['AND'].' '.$last;
56:         }
57: 
58:         return implode(' ',$aDate);
59:     }
