 48:     /**
 49:     *  The routing system.
 50:     *  <p>In this context a 'segment' of URI is a one of many parts separated by '/'.</p>
 51:     *  <p>In example '/my/route/test' there are three segments.</p>
 52:     *  <pre>
 53:     *  The route processing obey the follow sequence jumping to the next when not found:
 54:     *      1º - Search for direct route; (without URI params);
 55:     *      2º - Search if exist a method in controller of route '/' (with URI params);
 56:     *      3º - Search for controller/action/params name (with URI params);
 57:     *      4º - Search for auto detect route (with URI params);
 58:     *      5º - Search for named params in route;
 59:     *      6º - Load Not Found Controller;
 60:     *
 61:     *  Route format:   $route[URI][METHOD] = Array('ControllerName', 'Action', 'ModuleName');
 62:     *                  $route[URI][METHOD] = '/othet/route';
 63:     *                  METHOD can be: *, get, post, put, delete;
 64:     *                  When to use *, the others for the same URI will be skipped;
 65:     *  uri             => the relative path of resource like: /article/firstArticle ;
 66:     *  ControllerName  => the file name (without extension) of controller inside 'controller/' dir;
 67:     *  Action          => the method name of controller;
 68:     *                     if not specified, means that the action will be a method from Controller ;
 69:     *  ModuleName      => the module name inside 'APP_PATH/module/' dir;
 70:     *                     it means that the controller will be called from this module;
 71:     *  </pre>
 72:     *  Route examples:
 73:     *  <code>
 74:     *      // Module 'Admin' has controller 'UserAdmin' with method 'index'
 75:     *      $route['/admin/user']['*']   = Array('UserAdmin', 'index', 'Admin');
 76:     *
 77:     *      // Controller 'Home' has 'index' method (implicit)
 78:     *      $route['/about']['*']        = Array('Home');
 79:     *
 80:     *      // Controller 'Blog' has 'index' method
 81:     *      $route['/blog']['*']         = Array('Blog', 'index');
 82:     *  </code>
 83:     *
 84:     *  There are a optional behaviour when an additional segments in a URI are put as argument to the method called, example:
 85:     *  <code>
 86:     *      $route['/about']['*']        = Array('Home');
 87:     *      // if called URI '/about/John'
 88:     *      // the part URI (segment) 'John' will be passed as argument to method when the programmer can catch this value
 89:     *      function about($name) {
 90:     *          echo $name;
 91:     *      }
 92:     *
 93:     *  </code>
 94:     *  In this case method param $name is optional, if you need the value inside of called method.
 95:     *
 96:     *  Avoid this routes (don't works or works the wrong way):<code>
 97:     *    /blog/:id::name:   -> params together;
 98:     *    /blog/post/:post:  -> get all, but depends of declared order;
 99:     *    /blog/post/a:post: -> the previous route get this pattern;
100:     *    /blog/a:id:-:month:  -> allowed only one param per segment;</code>
101:     *
102:     *  Take care with routes that have same segments in URI, like:
103:     *  <code>
104:     *      $route['/']['*']          = Array('Main');
105:     *      $route['/buy/cpu']['*']   = Array('Buy');
106:     *
107:     *      // If controller 'Main' has 'buy' method, this route won't work as you think
108:     *      $route['/buy/:component:']['*']  = Array('Components', 'buy');
109:     *  </code>
110:     *
111:     *    @var Array
112:     */
113:     public    static $aRoute    = Array();

289:         $ret = true;
290:         if ( is_array($className) ) {
291:             $out   = Array();
292:             $class = current($className);
293:             do  {
294:                 $out[] = self::load($class, $path, $createObj);
295:             } while ( ($class=next($className)) );
296:             return $out;
297:         }
298:         if ( !class_exists($className)  ) {
299:             if ( is_file($path . $className.'.php') ) {
300:                 $ret =  (include $path . $className.'.php')!=1?false:true;
301:             } else {
302:                 $ret = false;
303:             }
304:         }
305: 
306:         if ( $createObj ) {
307:             return new $className;
308:         }
309:         return $ret;
310:     }

453:     /**
454:     *   Get named params from uri.
455:     *   @param Array $rule
456:     *   @param Array $data
457:     *   @return Array - Array('paramName' => 'value')
458:     */
459:     public static function getParams($rule, $data) {
460:         // remove empty items
461:         $ruleItems = array_filter( explode('/',$rule) );
462:         $dataItems = array_filter( explode('/',$data) );
463: 
464:         $params = array();
465: 
466:         foreach($ruleItems as $ruleKey => $ruleValue) {
467:             $i = strpos($ruleValue, ':');
468:             if  ( $i === false ) { continue; }
469:             $size = strlen($ruleValue);
470: 
471:             // check if the chunk is a key
472:             if (preg_match('/(:[a-zA-Z0-9]+:)/',$ruleValue)) {
473:                 $f = strpos($ruleValue, ':', $i+1);
474:                 if ( $i === 0 ) {// start at begin
475:                     // ended before end
476:                     if ( $f < $size-1 ) {
477:                         $te = strlen($dataItems[$ruleKey]) - ($size - $f-1);
478:                         $params[substr($ruleValue, $i+1, $f-$i-1)] = substr($dataItems[$ruleKey],0,$te);
479:                     } else {
480:                         $params[substr($ruleValue, $i+1, $f-1)] = $dataItems[$ruleKey];;
481:                     }
482: 
483:                 } else { //when not start at begin
484:                     // ended before end
485:                     if ( $f < $size-1 ) {
486:                         $te = strlen($dataItems[$ruleKey])-($i+($size-($f+1)));
487:                         $params[substr($ruleValue, $i+1, $f-$i-1)] = substr($dataItems[$ruleKey],$i, $te);;
488:                     } else {
489:                         $params[substr($ruleValue, $i+1, $f-1)] = substr($dataItems[$ruleKey],$i);;
490:                     }
491:                 }
492:             }
493:         }
494:         return $params;
495:     }

524:     *   @param Array  $aRoute
525:     *   @param String $uri
526:     *   @return boolean
527:     */
528:     protected static function parseRouteParams($aRoute, $uri) {
529: 
530:         foreach ($aRoute as $km => $vm)  {
531:             if ( strpos($km, ':')===false ) { continue ;}
532:             if (  substr_count($km, '/') != substr_count($uri, '/') )  { continue; }
533:             $regex = preg_replace('/(:[a-zA-Z0-9]+:)/', '(.+)', $km);
534:             $regex = str_replace('/','\/', $regex);
535:             // there are some route that match ?
536:             if ( preg_match('/('.$regex.')/', $uri) ) {
537:                 $out['route']  = $km;
538:                 $out['params'] = self::getParams($km, $uri) ;
539:                 return $out;
540:             }
541:         }
542:         return false;
543:     }

616:         // get URI
617:         $aUrl = parse_url($_SERVER['REQUEST_URI']);
618:         if ( ($indexPos = strpos($aUrl['path'], '/index.php')) === false ) { // don't have index.php
619:             if ( $_SERVER['SCRIPT_NAME'] != '/index.php' )  { // has subdir
620:                 $uri = str_replace( dirname($_SERVER['SCRIPT_NAME']), '', $aUrl['path']);
621:             } else {
622:                 $uri = &$aUrl['path'];
623:             }
624:         } else {
625:             $uri = '/'.substr($aUrl['path'], $indexPos+11);
626:         }
627:         ( isset($uri[1]) and substr($uri, -1)=='/')?($uri[strlen($uri)-1]=''):false;
628:         $uri = trim($uri);
629:

631:         if ( !$uri_parts ) { $uri_parts = Array('',''); }
632: 
633:         // Direct match, load pre-configured route (the fast way, recommended)
634:         $routeOption = Liber::getRouteMethod($uri);
635:         if ( !self::processController( $routeOption[0], (isset($routeOption[1])?$routeOption[1]:'index'), (isset($routeOption[2])?$routeOption[2]:'') ) ) {
636:             // Try '/' as a main controller
637:             $routeOption = Liber::getRouteMethod('/');
638:             if ( !self::processController( $routeOption[0], $uri_parts[0], (isset($routeOption[2])?$routeOption[2]:''), array_slice($uri_parts, 1) ) ) {
639:                 // Try /controller/action/param1/param2...
640:                 if ( !self::processController( $uri_parts[0], (isset($uri_parts[1])?$uri_parts[1]:'index'), '', array_slice($uri_parts, 2)) ) {

645:                         $uri_detect .= '/'.$uri_parts[$i];
646:                         $routeOption = Liber::getRouteMethod( $uri_detect );
647:                         if ( isset( $uri_parts[($i+1)] )  and ($uri_found = self::processController( $routeOption[0], $uri_parts[($i+1)], (isset($routeOption[2])?$routeOption[2]:''), array_slice($uri_parts, ($i+2)) )) ) {
648:                             return;
649:                         }
