cache() 

Uses: ->cache(); // return cache expire time in seconds ->cache(true); // enable caching to default value 3600s ->cache(false); // disable caching ->cache('filename.html'); // return current Array data about the specified file ->cache('filename.html', true); // set caching to default value for specified file ->cache('filename.html', false); // disable caching for specified file ->cache('filename.html', 2000); // set caching with 2000s to specific file
